1.3. Acesso

Para acessar um item no vetor usamos [ e ]: 

notas[1] = 5.8;

Primeiro elemento: índice 0.

1.4. Definição com valores

tipo nome[tamanho] = { lista_de_valores };

double notas[4] = { 7.3, 5.8, 9.2, 8.5 };

Menos itens que a quantidade: recebem o valor 0. 

double notas[4] = { 7.3, 5.8 };

Mais itens: erro! 

double notas[4] = { 7.3, 5.8, 9.2, 8.5, 6.7 };

Opção sem tamanho: 

double notas[] = { 7.3, 5.8, 9.2, 8.5, 6.7 };

2. Constantes

Para usar um valor em vários lugares sem alteração, use uma CONSTANTE: 

#define NOME VALOR

Onde aparecer NOME no código, será substituído pelo VALOR. A troca é feita antes da compilação. 

#define TAMANHO 4

#include <stdio.h>
#define TAMANHO 4

int main(void) {
  double notas[TAMANHO];

  notas[0] = 7.3;
  notas[1] = 5.8;
  notas[2] = 9.2;
  notas[3] = 8.5;

  for(int i = 0; i < TAMANHO; i++) {
    printf("Nota: %f.\n", notas[i]);
  }

  double total = notas[0] + notas[1] + notas[2] + notas[3];
  double media = total / 4.0;

  printf("Media: %f.\n", media);
}

Constantes são simbólicas, não podem receber valores. 

TAMANHO = 3; /* erro! */

4. Vetor como parâmetro

Ao chamar uma função com o nome do vetor como argumento, passamos o endereço do primeiro elemento. 

int resultado = soma(vetor, 2);

Para verificar podemos usar %p para imprimir o endereço na memória: 

#include <stdio.h>

int main(void) {
  int v[2] = { 3, 4 };

  printf("Endereço de v: %p\n", &v);
  printf("Endereço do v[0]: %p\n", &v[0]);
  printf("Nome de v: %p\n", v);	   
  return 0;
}

Assim, o nome do vetor é sempre o endereço do primeiro elemento. 

#include <stdio.h>

int soma(int n[], int tamanho);

int main() {
  int vetor[2] = { 7, 8 };

  int resultado = soma(vetor, 2);

  printf("Soma: %d\n", resultado);
}

int soma(int n[], int tamanho) {
  int total = 0;

  for(int i = 0; i < tamanho; i++) {
    total += n[i];
  }
  return total;
}

4.2. Relembre a passagem por valor e por referência:

#include <stdio.h>

void por_valor(int x);

void por_referencia(int *x);

int main(void) {
  int y = 3;

  printf("Y antes: %d\n", y);
  por_valor(y);
  printf("Y depois: %d\n\n", y);

  printf("Y antes: %d\n", y);
  por_referencia(&y);
  printf("Y depois: %d\n\n", y);

  return 0;
}

void por_referencia(int *x) {
  printf("X antes: %d\n", *x);
  *x *= 3;
  printf("X depois: %d\n", *x);
}

void por_valor(int x) {
  printf("X antes: %d\n", x);
  x *= 3;
  printf("X depois: %d\n", x);
}

A passagem do vetor para uma função é por referência e não por valor: 

#include <stdio.h>

void altera(int v[]);

int main(void) {
  int vet[] = { 7, 8, 9 };

  puts("Original: ");
  for( int i = 0; i < 3; i++ ) {
    printf("%d ", vet[i]);
  }

  altera(vet);

  puts("\nAlterado: ");
  for( int i = 0; i < 3; i++ ) {
    printf("%d ", vet[i]);
  }

  return 0;
}

void altera(int v[]) {
  for(int i = 0; i < 3; i++ ) {
    v[i] += 3;
  }
}
